Function 游戏登录(index, 自动登录)
    Dim 账号, 密码, 用户名,角色名
    Dim path, content, accounts, account, subAccounts, subAccount, i, j, currentIndex
    Dim 当前时间, 更新日期, 更新小时, 是否需要登录
    Dim 更新日期时间

    // 1. 读取账号.json文件
    path = GetSdcardDir() & "/账号.json"
    content = file.Read(path)
    If content = "" Or content = Null Then
        TracePrint "未找到账号配置文件或文件为空"
        Exit Function
    End If
    // 2. 解析JSON
    accounts = Encode.JsonToTable(content)
    If accounts = Null Then
        TracePrint "账号配置解析失败"
        Exit Function
    End If

    // 3. 获取当前网络时间
    当前时间 = GetNetworkTime()
    TracePrint "当前网络时间：" & 当前时间

    // 4. 判断是否为自动登录模式
    If 自动登录 = True Then
        TracePrint "自动登录模式：从第一个账号开始检查更新时间"
        // 遍历所有账号，找到第一个需要登录的账号
        For i = 1 To Len(accounts)
            account = accounts[i]
            更新日期 = account["更新时间"]
            TracePrint "检查第 " & i & " 个账号，更新时间：" & 更新日期

            // 判断更新时间是否在今天服务器刷新时间（4:00）之前
            // 解析更新时间
            If 更新日期 <> "" And 更新日期 <> Null Then
                // 提取日期部分（YYYY-MM-DD）
                更新日期时间 = Left(更新日期, 10)
                // 提取小时部分
                更新小时 = Mid(更新日期, 12, 2)
                更新小时 = CInt(更新小时)

                // 判断逻辑：
                // 1. 如果更新日期是今天，且更新时间在4:00之后，说明今天已登录
                // 2. 如果更新日期是今天之前，说明昨天或更早登录
                // 3. 如果更新日期是今天且在4:00之前，说明需要登录
                If Left(当前时间, 10) = 更新日期时间 Then
                    // 更新日期是今天
                    If 更新小时 >= 4 Then
                        TracePrint "账号今日已登录，跳过"
                        是否需要登录 = False
                    Else
                        TracePrint "账号今日4点前登录，需要重新登录"
                        是否需要登录 = True
                    End If
                Else
                    // 更新日期不是今天，需要登录
                    TracePrint "账号更新日期不是今天，需要登录"
                    是否需要登录 = True
                End If
            Else
                // 没有更新时间记录，需要登录
                TracePrint "账号无更新时间记录，需要登录"
                是否需要登录 = True
            End If

            If 是否需要登录 = True Then
                TracePrint "找到需要登录的账号，index：" & i
                index = i
                Exit For
            End If
        Next

        // 如果遍历完所有账号都没有找到需要登录的，返回空
        If 是否需要登录 = False Then
            TracePrint "所有账号今日已完成登录，无需操作"
            Exit Function
        End If
    End If

    // 5. 查找对应index的小号
    // 由于JSON结构是扁平的数组，直接根据index获取即可
    If index > 0 And index <= Len(accounts) Then
        account = accounts[index]

        账号 = account["账号"]
        密码 = account["密码"]
        用户名 = account["用户名"]
        角色名=account["角色名"]

        TracePrint "找到第 " & index & " 个小号信息："
        TracePrint "账号: " & 账号
        // TracePrint "密码: " & 密码
        TracePrint "用户名: " & 用户名

        Call 执行登录操作(账号, 密码, 用户名, 角色名)

        Exit Function
    Else
        TracePrint "未找到第 " & index & " 个小号，索引超出范围"
    End If
End Function


Function 执行登录操作(账号, 密码, 用户名,角色名)
    Dim 开始时间, 界面状态
    开始时间 = TickCount()
    TracePrint "开始执行登录操作，账号：" & 账号 & "，用户名：" & 用户名 & "，角色名：" & 角色名
    If Sys.IsRunning("com.df.dldldjbzf.sj") = True Then
        KillApp "com.df.dldldjbzf.sj"
        Delay 1000
    End If
    RunApp "com.df.dldldjbzf.sj"
    Delay 1000

    // 等待登录界面，同时判断是"已登录"还是"未登录"状态
    界面状态 = ""
    Do
        If TickCount() - 开始时间 > 30000 Then
            Call 发送PushPlus(g_pushplus_token, "登录失败", "等待登录界面超时:" & 账号)
            Exit Function
        End If
        TracePrint "检测登录界面状态..."
        // 检测未登录状态：确认登录按钮
        If 区域是否包含指定文字(g_xy["登录界面确认密码登录"]["x1"], g_xy["登录界面确认密码登录"]["y1"], g_xy["登录界面确认密码登录"]["x2"], g_xy["登录界面确认密码登录"]["y2"], "密码登录") Then
            TracePrint "检测到确认登录界面，当前为未登录状态"
            界面状态 = "未登录"
            Exit Do
        End If
        // 检测已登录状态：增加小号文字
        If 区域是否包含指定文字(g_xy["登录界面增加小号文字"]["x1"], g_xy["登录界面增加小号文字"]["y1"], g_xy["登录界面增加小号文字"]["x2"], g_xy["登录界面增加小号文字"]["y2"], "增加小号") Then
            TracePrint "检测到增加小号文字，当前为已登录状态"
            界面状态 = "已登录"
            Exit Do
        End If
        Delay 500
    Loop

    If 界面状态 = "未登录" Then
        // 未登录：跳过找切换账号按钮，直接输入账号密码
        TracePrint "执行未登录账号流程，输入账号密码登录..."
        If Not 切换并登录账号(账号, 密码, 开始时间, True) Then
            Call 发送PushPlus(g_pushplus_token, "登录失败", "切换账号失败:" & 账号)
            Exit Function
        End If
        // 登录成功后，重新等待登录界面出现（此时应为新账号的已登录状态）
        TracePrint "登录成功，等待登录界面出现..."
        开始时间 = TickCount()
        Do
            If TickCount() - 开始时间 > 30000 Then
                Call 发送PushPlus(g_pushplus_token, "登录失败", "登录后等待界面超时:" & 账号)
                Exit Function
            End If
            TracePrint "检测登录后界面状态..."
            If 区域是否包含指定文字(g_xy["登录界面确认登录"]["x1"], g_xy["登录界面确认登录"]["y1"], g_xy["登录界面确认登录"]["x2"], g_xy["登录界面确认登录"]["y2"], "确认登录") Then
                TracePrint "登录后仍为未登录状态，可能登录未成功"
                Exit Function
            End If
            If 区域是否包含指定文字(g_xy["登录界面增加小号文字"]["x1"], g_xy["登录界面增加小号文字"]["y1"], g_xy["登录界面增加小号文字"]["x2"], g_xy["登录界面增加小号文字"]["y2"], "增加小号") Then
                TracePrint "登录成功，检测到已登录状态"
                Exit Do
            End If
            Delay 500
        Loop
    End If
    // 已登录：判断当前账号是否为目标账号（未登录执行完切换后也走这里）
    Do
        If TickCount() - 开始时间 > 30000 Then
            Call 发送PushPlus(g_pushplus_token, "登录失败", "选择账号超时:" & 账号)
            Exit Function
        End If
        TracePrint "检测当前账号是否为目标账号：" & 账号
        If 区域是否包含指定文字(g_xy["登录界面当前账号"]["x1"], g_xy["登录界面当前账号"]["y1"], g_xy["登录界面当前账号"]["x2"], g_xy["登录界面当前账号"]["y2"], 账号) Then
            TracePrint "当前账号已是目标账号"
            Exit Do
        Else
            TracePrint "当前账号不是目标账号，切换已登录账号"
            If Not 切换已登录账号(账号, 开始时间) Then
                Call 发送PushPlus(g_pushplus_token, "登录失败", "切换账号失败:" & 账号)
                Exit Function
            End If
        End If
        Delay 500
    Loop

    If Not 选择用户名(用户名, 开始时间) Then
        Call 发送PushPlus(g_pushplus_token, "登录失败", "未找到用户名:" & 用户名)
        Exit Function
    End If
    If Not 处理公告并进入游戏(角色名, 开始时间) Then
        Call 发送PushPlus(g_pushplus_token, "登录失败", "进入游戏失败:" & 角色名)
        Exit Function
    End If
End Function
// 循环识别区域文字是否匹配
Function 循环检测目标文字(x1, y1, x2, y2, text, totle, times)
    Dim 次数, 目标文字
    次数 = totle
    循环检测目标文字 = False
    While 次数 > 0
        目标文字 = 识别区域文字(x1, y1, x2, y2)
        If 目标文字 <> -1 Then 
            目标文字 = 目标文字(0)
            目标文字 = 目标文字["text"]
            TracePrint "当前文字：" & 目标文字
            If 目标文字 = text Then 
                循环检测目标文字 = True
                Exit Function
            End If
        End If
        次数 = 次数 - 1
        If 次数 > 0 Then
            Delay times
        End If
    Wend
    TracePrint "出错了"
End Function

Function 发送PushPlus(token, title, content)
    Dim apiUrl, postData, resp
    apiUrl = "https://www.pushplus.plus/send/"
    postData = "token=" & token & "&title=" & title & "&content=" & content
    resp = url.post(apiUrl, postData)
    发送PushPlus = resp
    TracePrint "PushPlus返回：" & resp
End Function

Function 选择用户名(用户名, 开始时间)
    Dim 滑动次数
    选择用户名 = False
    TracePrint "开始选择用户名：" & 用户名
    滑动次数 = 0
    Do While 滑动次数 <= 2
        If TickCount() - 开始时间 > 30000 Then
            Exit Function
        End If
        If 在区域点击指定文字(g_xy["登录界面寻找用户名"]["x1"], g_xy["登录界面寻找用户名"]["y1"], g_xy["登录界面寻找用户名"]["x2"], g_xy["登录界面寻找用户名"]["y2"], 用户名) Then
            TracePrint "已选择用户名：" & 用户名
            选择用户名 = True
            Exit Function
        End If
        If 滑动次数 = 2 Then
            TracePrint "已滑动两次仍未找到用户名：" & 用户名
            Exit Do
        End If
        TracePrint "未找到用户名，执行滑动，当前滑动次数：" & 滑动次数
        Call 按配置滑动("登录界面寻找用户名滑动")
        滑动次数 = 滑动次数 + 1
        Delay 800
    Loop
    TracePrint "选择用户名失败：" & 用户名
End Function

            TracePrint "检测到角色名：" & 角色名 & "，进入游戏成功"
            处理公告并进入游戏 = True
            Exit Function
        End If
        If 区域是否包含指定文字(g_xy["登录界面公告"]["x1"], g_xy["登录界面公告"]["y1"], g_xy["登录界面公告"]["x2"], g_xy["登录界面公告"]["y2"], "公告") Then
            TracePrint "检测到公告界面，执行关闭公告并进入游戏" & g_xy["登录界面关闭公告"]["x1"], g_xy["登录界面关闭公告"]["y1"]
            Tap g_xy["登录界面关闭公告"]["x1"], g_xy["登录界面关闭公告"]["y1"]
            Delay 1500
            // 等待进入游戏按钮出现
            Do
                If TickCount() - 开始时间 > 30000 Then
                    TracePrint "等待进入游戏按钮超时"
                    Exit Function
                End If
                If 区域是否包含指定文字(g_xy["登录界面进入游戏按钮判断"]["x1"], g_xy["登录界面进入游戏按钮判断"]["y1"], g_xy["登录界面进入游戏按钮判断"]["x2"], g_xy["登录界面进入游戏按钮判断"]["y2"], "进入游戏") Then
                    TracePrint "检测到进入游戏按钮"
                    Exit Do
                End If
                Delay 500
            Loop
            // 判断用户协议颜色
            Dim 协议颜色
            协议颜色 = GetPixelColor(g_xy["登录界面用户协议"]["x1"], g_xy["登录界面用户协议"]["y1"], 0)
            TracePrint "当前用户协议颜色：" & 协议颜色
            If 协议颜色 <> "51C05A" Then
                TracePrint "用户协议未勾选，点击勾选"
                Tap g_xy["登录界面用户协议"]["x1"], g_xy["登录界面用户协议"]["y1"]
                Delay 500
                协议颜色 = GetPixelColor(g_xy["登录界面用户协议"]["x1"], g_xy["登录界面用户协议"]["y1"], 0)
                TracePrint "勾选后用户协议颜色：" & 协议颜色
            End If
            If 协议颜色 = "51C05A" Then
                TracePrint "用户协议已勾选，点击进入游戏"
                Tap g_xy["登录界面进入游戏"]["x1"], g_xy["登录界面进入游戏"]["y1"]
                Delay 1000
            End If
        End If
        Delay 500
    Loop
End Function
