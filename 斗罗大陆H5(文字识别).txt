// 加载分辨率配置文件
// 功能: 删除旧文件、释放附件、读取并打印内容
// 返回值: 文件内容字符串
Function LoadResolutionFile()
    Dim path, content
    path = GetSdcardDir() & "/分辨率坐标.json"
    
    // 1. 检测并删除旧文件
    // 注意: 不同版本的按键精灵可能有不同的文件检测命令，这里使用通用的 File.Exist
    If dir.Exist(path) Then 
        dir.Delete(path)
        TracePrint "旧配置文件已删除: " & path
    End If
    
    // 2. 释放附件中的配置文件
    // PutAttachment 释放附件到指定目录
    // 参数1: 目标目录, 参数2: 附件文件名 (可选, 不填则释放所有)
    PutAttachment GetSdcardDir(), "分辨率坐标.json"
    TracePrint "尝试释放配置文件到: " & path
    
    // 3. 读取文件内容
    content = File.Read(path)
    
    // 4. 打印文件内容
    If content <> "" And content <> Null Then
        TracePrint "配置文件内容读取成功："
        TracePrint content
    Else
        TracePrint "配置文件内容为空或读取失败！请检查附件中是否存在 分辨率坐标.json"
    End If
    
    LoadResolutionFile = content
End Function

// 初始化分辨率配置
// 参数: w - 宽度, h - 高度
// 返回值: 配置表或Null
Function InitResolution(w, h)
    Dim content
    content = LoadResolutionFile()
    Dim config_table, res_key, res_config
    // 尝试解析配置文件
    If content <> "" And content <> Null Then
        // 注意：Encode.JsonToTable 是按键精灵常用的 JSON 解析函数
        // 如果环境不支持，可能需要手动解析或使用正则表达式
        config_table = Encode.JsonToTable(content)
        res_key = w & "*" & h
        
        If config_table[res_key] Then
            TracePrint "找到分辨率配置: " & res_key
            res_config = config_table[res_key]
            
            // 尝试从配置中读取坐标 (如果存在则覆盖默认值)
            // 示例映射：将 JSON 中的 "登录按钮" 映射到 g_login_button (需自行定义)
            // 这里仅作演示，实际需根据 JSON 键名修改
            InitResolution = res_config
        Else
            TracePrint "配置文件中未找到分辨率: " & res_key & "，使用默认配置"
            InitResolution = Null
        End If
    Else
        InitResolution = Null
    End If
End Function

// 循环识别区域文字是否匹配
Function 循环检测目标文字(x1, y1, x2, y2, text, totle, times)
    Dim 次数, 目标文字
    次数 = totle
    循环检测目标文字 = False
    While 次数 > 0
        目标文字 = 识别区域文字(x1, y1, x2, y2)
        If 目标文字 <> -1 Then 
            目标文字 = 目标文字(0)
            目标文字 = 目标文字["text"]
            TracePrint "当前文字：" & 目标文字
            If 目标文字 = text Then 
                循环检测目标文字 = True
                Exit Function
            End If
        End If
        次数 = 次数 - 1
        If 次数 > 0 Then
            Delay times
        End If
    Wend
    TracePrint "出错了"
End Function

//识别区域文字
Function 识别区域文字(x1, y1, x2, y2)
    dim ret = ocrEx(x1, y1, x2, y2)
    If Len(ret) = 1 and ret[1]["text"] = "" Then 
        识别区域文字=-1
    Else 
        识别区域文字=ret
    End If
End Function
// 打印目标区域文字
Function 打印目标区域文字(x1,y1,x2,y2)
    Dim ret, i
    ret = OcrEx(x1, y1, x2, y2)
    If Len(ret) = 1 and ret[1] = "" Then
        TracePrint "没有识别到文本"
    Else
        For i = 1 to Len(ret)
            TracePrint "识别到目标：" & ret[i]["text"] & _
                ",左上角x坐标：" & ret[i]["x"] & _
                ",左上角y坐标：" & ret[i]["y"] & _
                ",区域宽：" & ret[i]["w"] & _
                ",区域高：" & ret[i]["h"] & _
                ",可信度：" & ret[i]["score"]
        Next
    End If
    打印目标区域文字 = ret
End Function

Function 发送PushPlus(token, title, content)
    Dim apiUrl, postData, resp
    apiUrl = "https://www.pushplus.plus/send/"
    postData = "token=" & token & "&title=" & title & "&content=" & content

End Function
