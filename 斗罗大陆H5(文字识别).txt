Function 游戏登录(index)
    Dim 账号, 密码, 用户名,角色名
    Dim path, content, accounts, account, subAccounts, subAccount, i, j, currentIndex
    
    // 1. 读取账号.json文件
    path = GetSdcardDir() & "/账号.json"
    content = file.Read(path)
    If content = "" Or content = Null Then
        TracePrint "未找到账号配置文件或文件为空"
        Exit Function
    End If
    // 2. 解析JSON
    accounts = Encode.JsonToTable(content)
    If accounts = Null Then
        TracePrint "账号配置解析失败"
        Exit Function
    End If
    
    // 3. 查找对应index的小号
    // 由于JSON结构是扁平的数组，直接根据index获取即可
    If index > 0 And index <= Len(accounts) Then
        account = accounts[index]
        
        账号 = account["账号"]
        密码 = account["密码"]
        用户名 = account["用户名"]
        角色名=account["角色名"]
        
        TracePrint "找到第 " & index & " 个小号信息："
        TracePrint "账号: " & 账号
        // TracePrint "密码: " & 密码 
        TracePrint "用户名: " & 用户名
        
        Call 执行登录操作(账号, 密码, 用户名, 角色名)
        
        Exit Function
    Else
        TracePrint "未找到第 " & index & " 个小号，索引超出范围"
    End If
End Function

// 循环识别区域文字是否匹配
Function 循环检测目标文字(x1, y1, x2, y2, text, totle, times)
    Dim 次数, 目标文字
    次数 = totle
    循环检测目标文字 = False
    While 次数 > 0
        目标文字 = 识别区域文字(x1, y1, x2, y2)
        If 目标文字 <> -1 Then 
            目标文字 = 目标文字(0)
            目标文字 = 目标文字["text"]
            TracePrint "当前文字：" & 目标文字
            If 目标文字 = text Then 
                循环检测目标文字 = True
                Exit Function
            End If
        End If
        次数 = 次数 - 1
        If 次数 > 0 Then
            Delay times
        End If
    Wend
    TracePrint "出错了"
End Function

//识别区域文字
Function 识别区域文字(x1, y1, x2, y2)
    dim ret = ocrEx(x1, y1, x2, y2)
    If Len(ret) = 1 and ret[1]["text"] = "" Then 
        识别区域文字=-1
    Else 
        识别区域文字=ret
    End If
End Function
// 打印目标区域文字
Function 打印目标区域文字(x1,y1,x2,y2)
    Dim ret, i
    ret = OcrEx(x1, y1, x2, y2)
    If Len(ret) = 1 and ret[1] = "" Then
        TracePrint "没有识别到文本"
    Else
        For i = 1 to Len(ret)
            TracePrint "识别到目标：" & ret[i]["text"] & _
                ",左上角x坐标：" & ret[i]["x"] & _
                ",左上角y坐标：" & ret[i]["y"] & _
                ",区域宽：" & ret[i]["w"] & _
                ",区域高：" & ret[i]["h"] & _
                ",可信度：" & ret[i]["score"]
        Next
    End If
    打印目标区域文字 = ret
End Function

Sub 滑动(起点X,起点Y,终点X,终点Y,滑动时间)
    TouchDown 起点X, 起点Y, 1
    Delay 200
    TouchMove 终点X, 终点Y, 1, 滑动时间

    tap 终点X, 终点Y

End Sub



Function 发送PushPlus(token, title, content)
    Dim apiUrl, postData, resp
    apiUrl = "https://www.pushplus.plus/send/"
    postData = "token=" & token & "&title=" & title & "&content=" & content
    resp = url.post(apiUrl, postData)
    发送PushPlus = resp
    TracePrint "PushPlus返回：" & resp
End Function
Function 区域是否包含指定文字(x1, y1, x2, y2, text)
    Dim ret, i
    TracePrint "开始检测区域是否包含指定文字，目标文字：" & text & "，区域：(" & x1 & "," & y1 & ")-(" & x2 & "," & y2 & ")"
    ret = 识别区域文字(x1, y1, x2, y2)
    区域是否包含指定文字 = False
    If ret = -1 Then
        TracePrint "区域识别结果为空"
        Exit Function
    End If
    For i = 1 To Len(ret)
    TracePrint ret[i]["text"]
        If ret[i]["text"] = text Then
            TracePrint "区域检测到目标文字：" & ret[i]["text"]
            区域是否包含指定文字 = True
            Exit Function
        End If
    Next
    TracePrint "区域未检测到目标文字：" & text
End Function
Function 在区域点击指定文字(x1, y1, x2, y2, text)
    Dim ret, i
    TracePrint "开始在区域内查找并点击指定文字，目标文字：" & text & "，区域：(" & x1 & "," & y1 & ")-(" & x2 & "," & y2 & ")"
    ret = 识别区域文字(x1, y1, x2, y2)
    在区域点击指定文字 = False
    If ret = -1 Then
        TracePrint "区域识别结果为空，无法点击"
        Exit Function
    End If
    For i = 1 To Len(ret)
        TracePrint ret[i]["text"]
        If InStr(1, ret[i]["text"], text) > 0 Then
            TracePrint "点击匹配到的文字：" & ret[i]["text"] & "，坐标：" & ret[i]["x"] & "," & ret[i]["y"]
            Tap ret[i]["x"], ret[i]["y"]
            在区域点击指定文字 = True
            Exit Function
        End If
    Next
    TracePrint "区域内未找到可点击的目标文字：" & text
End Function

Function 选择用户名(用户名, 开始时间)
    Dim 滑动次数
    选择用户名 = False
    TracePrint "开始选择用户名：" & 用户名
    滑动次数 = 0
    Do While 滑动次数 <= 2
        If TickCount() - 开始时间 > 30000 Then
            Exit Function
        End If
        If 在区域点击指定文字(g_xy["登录界面寻找用户名"]["x1"], g_xy["登录界面寻找用户名"]["y1"], g_xy["登录界面寻找用户名"]["x2"], g_xy["登录界面寻找用户名"]["y2"], 用户名) Then
            TracePrint "已选择用户名：" & 用户名
            选择用户名 = True
            Exit Function
        End If
        If 滑动次数 = 2 Then
            TracePrint "已滑动两次仍未找到用户名：" & 用户名
            Exit Do
        End If
        TracePrint "未找到用户名，执行滑动，当前滑动次数：" & 滑动次数
        Call 按配置滑动("登录界面寻找用户名滑动")
        滑动次数 = 滑动次数 + 1
        Delay 800
    Loop
    TracePrint "选择用户名失败：" & 用户名
End Function
